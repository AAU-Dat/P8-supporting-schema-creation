\section{Proof of \autoref{thm:galios}}\label{sec:galois}

\galois

\begin{proof}
    \pf\
    \step{meet-morphism-4}{$\concrete_4$ is a meet morphism.}
    \begin{proof}
        \pf\
        \begin{align}
            \concrete_4 \left( \bigcap_{i = 1}^n R_i \right)
            &=  \mathcal{L} \left( \bigcap_{i = 1}^n R_i \right)\\
            &=  \bigcap_{i = 1}^n \mathcal{L} \left( R_i \right)\\
            &=  \bigcap_{i = 1}^n \concrete_4 \left( R_i \right)
        \end{align}
    \end{proof}
    \step{meet-morphism-3}{$\concrete_3$ is a meet morphism.}
    \begin{proof}
        \pf\
        \begin{align*}
            \concrete_3 \left( \bigsqcap_{i = 1}^m (e_{i1}^\#, e_{i2}^\#, \dots, e_{in}) \right)
            &= \concrete_3 \left( \bigsqcap_{i = 1}^m e_{i1}^\#, \bigsqcap_{i = 1}^m e_{i2}^\#, \dots, \bigsqcap_{i = 1}^m e_{in}^\# \right) \\
            &= \bigtimes_{j = 1}^n \concrete_4 \left( \bigsqcap_{i = 1}^m e_{ij}^\# \right) \\
            &= \bigtimes_{j = 1}^n \bigcap_{i = 1}^m \concrete_4 (e_{ij}^\#) \text{ as $\concrete_4$ is a meet morphism by \stepref{meet-morphism-4}} \\
            &= \bigcap_{i = 1}^m \bigtimes_{j = 1}^n \concrete_4 (e_{ij}^\#) \\
            &= \bigcap_{i = 1}^m \concrete_3 (e_{i1}^\#, e_{i2}^\#, \dots, e_{in}^\#) \\
        \end{align*}
    \end{proof}
